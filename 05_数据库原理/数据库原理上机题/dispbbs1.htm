 并用SQL语言完成以下功能：
 1 建表，在定义中要求声明：
    （1）每个表的主外码。
    （2）每个班级的人数不能超过30人。

use pku2

create table c
(c# char(4),
 mname varchar(20),
 dname varchar(20),
 en_year char(4),
 pe_num tinyint,
 primary key(c#),
 --foreign key(d#) references d(d#) on delete cascade,
 check(pe_num <= 30))       --班级人数不超过30人

create table s
(s# char(8),
 sname varchar(20) not null, --学生姓名不为空
 age tinyint,
 c# char(4),
 primary key(s#),
 foreign key(c#) references c(c#) on delete cascade,
 check(age>=15 and age<=40))

create table d
(d# char(8),
 dname varchar(20),
 primary key(d#))
