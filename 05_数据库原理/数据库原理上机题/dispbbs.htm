 1Create.sql 分别创建下列三个表：

 学生(学号，姓名，性别，年龄，系别)
 课程(课程号，课程名，任课教师)

 并用SQL语言完成以下功能：

 1 建表，在定义中要求声明：
   (1)每个表的主外码；
   (2)学生的年龄介于16到40之间；
   (3)学生的姓名和课程名不能为空值；
   (4)选课成绩要么为空值，要么取0到100之间的整数。
 ************************************************/
use pku1
create table s
(s# varchar(8),
 sname varchar(20) not null,
 sex char(4) default '男',
 age tinyint,
 dname char(10)
 primary key (s#),
 check((sex = '男' or sex = '女') AND (age between 16 and 40)))

create table c
(c# varchar(20),
 cname varchar(20) not null,
 tname varchar(8)
 primary key(c#))

create table sc
(s# varchar(8),
 c# varchar(20),
 grade tinyint,
 primary key(s#, c#),
 foreign key(s#) references s(s#) on delete cascade,
 foreign key(c#) references c(c#) on delete cascade,
 check ((grade is NULL) or (grade between 0 and 100)))

/*****************************************
Query1.sql 该查询完成下列三个功能：

    (1)列出张三同学选修的所有课程的名称及其成绩。
    (2)列出所有课程都及格了的同学的名字。
    (3)列出在选修张青老师所教授课程的学生中，
       成绩最高的学生姓名及其成绩。
 *****************************************/

use pku1
-- (1)列出张三同学选修的所有课程的名称及其成绩。
select cname, grade
from s,c,sc
where sname = '张三' and s.s# = sc.s# and c.c# = sc.c#

-- (2)列出所有课程都及格了的同学的名字。
select sname
from s, sc
where s.s# = sc.s#
group by sname
having min(grade) >= 60

-- (3)列出在选修张青老师所教授课程的学生中
--    成绩最高的学生姓名及其成绩。
select sname, grade
from s, c, sc
where s.s# = sc.s# and grade = (
       select max(grade)
       from sc
       where tname = '张青' and sc.c# =c.c#)

--  5 将105号同学选修203号课程的成绩改为该门课程的平均成绩.
-- 做此问之前要先把4注释掉， 并恢复执行4问之前的数据库，
-- 因为4问把不及格的都删掉了， 而5问要操作的数据是105号同学

set grade = 
    (select avg(grade)
     from sc
     where c# = '203')
where s# = '105' and c# = '203'
