 数据库原理上机题目连载(二)[原创]
以下是北大去年12月数据库原理上机考试的第二题(原题):
[size=4]假定需要保存“图书”、“读者”、“借阅”和“作者”信息以及它们之间的关系。
其中图书有总编号、书名、作者、出版单位、单价等信息。读者有借书证号、姓名、性别、地址。作者有姓名、单位、性别、出生日期等信息。每位读者可以借阅多本书，每本书可以由多位读者借阅，借书时有借书日期。 
要求：
1. 设计数据库，建立合适的表来保存需要的信息。定义表的时候要给出主外码定义，书名、作者、读者姓名不能为空；性别只能取“男”或“女”；借书日期必须在1980年1月1日到2050年12月31日之间。
2. 往每个表中插入至少4条记录，并保证后面的查询结果不为空。
3. 找出借阅日期在2002年12月之前的读者名和书名。
4. 给出每个出版社出版的单价在20元以下的书的总数。
5. 作者“李通”现在改笔名为“李不通”，请在数据库内做出相应改变。
6. 删除所有在本库中没有其写的书的作者的信息。[/size]

以下是答案,已经编译通过:
drop table rb
drop table reader
drop table author
drop table book

/*1. 设计数据库*/
 /*图书有总编号、书名、作者、出版单位、单价等信息*/
create table book
(bno  char(4),
 bname char(10) not null,
 authorname char(6) not null,
 bplace char(10),
 price int,
 primary key (bno)
)
 /*读者有借书证号、姓名、性别、地址。*/
create table reader
(rno char(4),
 rname char(6) not null,
 sex char(2),
 address char(10),
 primary key (rno)
)
 /*作者有姓名、单位、性别、出生日期等信息。*/
create table author
(authorname char(6) not null,
 aplace char(10),
 sex char(2),
 birth int,
 primary key (authorname)
)
 /*借阅*/
create table rb
(rno char(4),
 bno char(4),
 time int,
 primary key (rno,bno),
 foreign key (rno) references reader(rno),
 foreign key (bno) references book(bno),
 check (time between 19800101 and 20501231)
)

/*2. 往每个表中插入至少4条记录，并保证后面的查询结果不为空。*/
insert into book values('b1','book1','李通','peking',20)
insert into book values('b2','book2','ath1','peking',16)
insert into book values('b3','book3','ath2','beikd',12)
insert into book values('b4','book4','李通','bgd',27)
insert into book values('b5','book5','ath3','bgd',12)
insert into book values('b6','book6','ath3','peking',11)
insert into book values('b7','book7','ath3','peking',10)

insert into reader values('r1','read1','男','nmg')
insert into reader values('r2','read2','女','tianjin')
insert into reader values('r3','read3','男','beijing')
insert into reader values('r4','read4','女','zj')

insert into author values('李通','beih','男',20001112)
insert into author values('ath1','peking','女',19680211)
insert into author values('ath2','bgd','男',19531123)
insert into author values('ath3','bkd','男',19781212)
insert into author values('ath4','blg','女',19800117)
insert into author values('ath5','blg','女',19810117)

insert into rb values('r1','b1',20000112)
insert into rb values('r1','b2',19991112)
insert into rb values('r2','b1',20030131)
insert into rb values('r2','b3',20000919)
insert into rb values('r2','b4',20030729)
insert into rb values('r3','b1',20001009)

/*3. 找出借阅日期在2002年12月之前的读者名和书名。*/
select rname,bname
from reader,book,rb
where reader.rno=rb.rno and book.bno=rb.bno and rb.time<20021200

/*4. 给出每个出版社出版的单价在20元以下的书的总数。*/ 
select bplace,count(*)
from book
where price<20
group by bplace

/*5. 作者“李通”现在改笔名为“李不通”，请在数据库内做出相应改变。*/
update author
set authorname='李不通'
where authorname='李通'
 
update book
set authorname='李不通'
where authorname='李通'

/*6. 删除所有在本库中没有其写的书的作者的信息。*/
delete from author
where authorname not in
     (select authorname 
      from book
      )
